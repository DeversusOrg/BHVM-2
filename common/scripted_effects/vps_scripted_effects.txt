check_vp_control = {
    if = {
        limit = {
            controls_state = var:captured_state
        }
        return = yes
    }
    else = {
        return = no
    }
}

check_pending_captures = {
    if = {
        limit = {
            has_variable = captured_state
            NOT = { controls_state = var:captured_state }
        }
        clear_variable = captured_state
    }
}
